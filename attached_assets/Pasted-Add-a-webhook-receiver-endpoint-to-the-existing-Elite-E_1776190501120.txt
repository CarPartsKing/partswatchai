STATUS_MAP = {

def parse_iso_time(ts):
    if not ts:
        return None
    try:
        dt = datetime.fromisoformat(ts.replace("Z", "+00:00"))
        return dt.strftime("%H:%M:%S")
    except:
        return None

def parse_iso_date(ts):
    if not ts:
        return None
    try:
        dt = datetime.fromisoformat(ts.replace("Z", "+00:00"))
        return dt.strftime("%Y-%m-%d")
    except:
        return None

def minutes_between(start_ts, end_ts):
    if not start_ts or not end_ts:
        return None
    try:
        fmt = "%Y-%m-%dT%H:%M:%SZ"
        s = datetime.strptime(start_ts.replace("+00:00","Z"), fmt)
        e = datetime.strptime(end_ts.replace("+00:00","Z"), fmt)
        diff = (e - s).total_seconds() / 60
        if 0 < diff < 720:
            return round(diff, 1)
        return None
    except:
        return None

def calc_on_time(eta_ts, delivered_ts):
    mins = minutes_between(eta_ts, delivered_ts)
    if mins is None:
        return 1
    return 1 if mins <= 15 else 0

def extract_stop_from_event(event):
    raw = event.get("raw_payload")
    if isinstance(raw, str):
        try:
            payload = json.loads(raw)
        except:
            return None
    else:
        payload = raw or {}

    status_raw = event.get("status") or payload.get("status", "")
    status = STATUS_MAP.get(status_raw, "delivered")

    if status_raw not in ("Delivered", "Arrived"):
        return None

    route_date = parse_iso_date(
        event.get("time_dispatched") or
        event.get("route_start_time") or
        event.get("time_created")
    )

    if not route_date:
        return None

    driver_name = (
        event.get("driver_full_name") or
        payload.get("driver_full_name") or
        "Unknown"
    ).strip()

    arrival_time   = parse_iso_time(event.get("arrival_time"))
    departure_time = parse_iso_time(event.get("departure_time"))
    dispatch_time  = parse_iso_time(event.get("time_dispatched"))
    delivered_time = parse_iso_time(event.get("time_delivered") or event.get("arrival_time"))
    scheduled_time = parse_iso_time(event.get("delivery_eta"))

    dwell_minutes             = minutes_between(event.get("arrival_time"), event.get("departure_time"))
    dispatch_to_delivery_min  = minutes_between(event.get("time_dispatched"), event.get("time_delivered") or event.get("arrival_time"))
    pick_to_dispatch_min      = None
    total_cycle_min           = dispatch_to_delivery_min

    on_time = calc_on_time(event.get("delivery_eta"), event.get("time_delivered") or event.get("arrival_time"))

    location = (
        event.get("store_id") or
        str(event.get("ordered_by_dealer_id") or "") or
        "webhook"
    )

    return {
        "route_id":                None,
        "import_id":               f"WEBHOOK-{event.get('manifest_id') or 'unknown'}",
        "route_date":              route_date,
        "driver_name":             driver_name,
        "stop_sequence":           event.get("stop_number"),
        "customer_name":           event.get("destination_business_name"),
        "customer_id":             event.get("destination_business_id"),
        "address":                 None,
        "order_number":            event.get("invoice"),
        "invoice_number":          event.get("invoice"),
        "order_value":             event.get("parts_value"),
        "dispatch_time":           dispatch_time,
        "scheduled_time":          scheduled_time,
        "arrival_time":            arrival_time,
        "departure_time":          departure_time,
        "delivered_time":          delivered_time,
        "status":                  status,
        "dwell_minutes":           dwell_minutes,
        "on_time":                 on_time,
        "pick_to_dispatch_min":    pick_to_dispatch_min,
        "dispatch_to_delivery_min": dispatch_to_delivery_min,
        "total_cycle_min":         total_cycle_min,
        "location":                location,
    }

def insert_stop_from_webhook(stop_dict):
    conn = get_connection()
    try:
        route_check = conn.execute("""
            SELECT id FROM routes
            WHERE route_date = ? AND driver_name = ? AND manifest_id = ?
            LIMIT 1
        """, [
            stop_dict["route_date"],
            stop_dict["driver_name"],
            stop_dict.get("import_id", "").replace("WEBHOOK-", "")
        ]).fetchone()

        route_id = route_check[0] if route_check else None

        if not route_id:
            result = conn.execute("""
                INSERT INTO routes (
                    import_id, route_date, driver_name, manifest_id,
                    location, dispatch_time, total_stops, delivered_stops,
                    attempted_stops, total_orders, total_miles, route_duration_min
                ) VALUES (?,?,?,?,?,?,0,0,0,0,0,0)
            """, [
                stop_dict["import_id"],
                stop_dict["route_date"],
                stop_dict["driver_name"],
                stop_dict.get("import_id", "").replace("WEBHOOK-",""),
                stop_dict["location"],
                stop_dict["dispatch_time"],
            ])
            route_id = result.lastrowid

        existing = conn.execute("""
            SELECT id FROM stops
            WHERE route_date = ? AND driver_name = ? AND invoice_number = ?
            LIMIT 1
        """, [
            stop_dict["route_date"],
            stop_dict["driver_name"],
            stop_dict["invoice_number"]
        ]).fetchone()

        if existing:
            conn.execute("""
                UPDATE stops SET
                    arrival_time = ?, departure_time = ?, delivered_time = ?,
                    status = ?, dwell_minutes = ?, on_time = ?,
                    dispatch_to_delivery_min = ?, total_cycle_min = ?
                WHERE id = ?
            """, [
                stop_dict["arrival_time"], stop_dict["departure_time"],
                stop_dict["delivered_time"], stop_dict["status"],
                stop_dict["dwell_minutes"], stop_dict["on_time"],
                stop_dict["dispatch_to_delivery_min"], stop_dict["total_cycle_min"],
                existing[0]
            ])
        else:
            conn.execute("""
                INSERT INTO stops (
                    route_id, import_id, route_date, driver_name, stop_sequence,
                    customer_name, customer_id, order_number, invoice_number,
                    order_value, dispatch_time, scheduled_time, arrival_time,
                    departure_time, delivered_time, status, dwell_minutes, on_time,
                    pick_to_dispatch_min, dispatch_to_delivery_min, total_cycle_min, location
                ) VALUES (?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?)
            """, [
                route_id, stop_dict["import_id"], stop_dict["route_date"],
                stop_dict["driver_name"], stop_dict["stop_sequence"],
                stop_dict["customer_name"], stop_dict["customer_id"],
                stop_dict["order_number"], stop_dict["invoice_number"],
                stop_dict["order_value"], stop_dict["dispatch_time"],
                stop_dict["scheduled_time"], stop_dict["arrival_time"],
                stop_dict["departure_time"], stop_dict["delivered_time"],
                stop_dict["status"], stop_dict["dwell_minutes"],
                stop_dict["on_time"], stop_dict["pick_to_dispatch_min"],
                stop_dict["dispatch_to_delivery_min"], stop_dict["total_cycle_min"],
                stop_dict["location"]
            ])

        conn.execute("""
            UPDATE routes SET
                total_stops = total_stops + 1,
                delivered_stops = delivered_stops + CASE WHEN ? = 'delivered' THEN 1 ELSE 0 END,
                attempted_stops = attempted_stops + CASE WHEN ? = 'attempted' THEN 1 ELSE 0 END
            WHERE id = ?
        """, [stop_dict["status"], stop_dict["status"], route_id])

        conn.commit()
        return True
    except Exception as e:
        conn.rollback()
        raise e
    finally:
        conn.close()

def process_pending_webhooks():
    events = get_unprocessed_webhook_events(limit=100)
    results = {"processed": 0, "skipped": 0, "failed": 0}

    for event in events:
        try:
            stop = extract_stop_from_event(event)
            if stop is None:
                mark_webhook_processed(event["id"])
                results["skipped"] += 1
                continue
            insert_stop_from_webhook(stop)
            mark_webhook_processed(event["id"])
            results["processed"] += 1
        except Exception as e:
            mark_webhook_processed(event["id"], error=str(e))
            results["failed"] += 1

    return results

pythondef verify_webhook_auth(request: Request) -> bool:
    auth_header = request.headers.get("Authorization", "")
    if not auth_header.startswith("Basic "):
        return False
    try:
        decoded = base64.b64decode(auth_header[6:]).decode("utf-8")
        username, password = decoded.split(":", 1)
        return username == WEBHOOK_USERNAME and password == WEBHOOK_PASSWORD
    except:
        return False

javascriptasync function loadWebhookStats() {
  try {
    const stats = await api('/api/webhook/stats');
    document.getElementById('wh-total').textContent     = stats.total_received || 0;
    document.getElementById('wh-processed').textContent = stats.total_processed || 0;
    document.getElementById('wh-pending').textContent   = stats.total_pending || 0;
    document.getElementById('wh-failed').textContent    = stats.total_failed || 0;

    const data = await api('/api/webhook/events?limit=20');
    const events = data.events || [];
    const list = document.getElementById('webhookEventsList');

    if (!events.length) {
      list.innerHTML = '<div style="color:#7a8099;padding:8px 0;">No webhook events received yet</div>';
      return;
    }

    list.innerHTML = events.map(e => {
      const statusColor = e.status === 'Delivered' ? '#22c55e' : e.status === 'Arrived' ? '#3b82f6' : '#7a8099';
      const procColor = e.processed === 1 ? '#22c55e' : e.processed === -1 ? '#ef4444' : '#f59e0b';
      const procLabel = e.processed === 1 ? 'ok' : e.processed === -1 ? 'failed' : 'pending';
      return `
        <div style="display:flex;align-items:center;justify-content:space-between;padding:7px 0;border-bottom:0.5px solid rgba(255,255,255,0.05);">
          <div>
            <span style="font-weight:500;color:#e8eaf0;">${e.driver_full_name || 'Unknown driver'}</span>
            <span style="color:#7a8099;margin-left:8px;">${e.destination_business_name || ''}</span>
          </div>
          <div style="display:flex;gap:10px;align-items:center;">
            <span style="color:${statusColor};font-size:11px;">${e.status || ''}</span>
            <span style="color:${procColor};font-size:10px;padding:1px 6px;border-radius:3px;background:rgba(255,255,255,0.05);">${procLabel}</span>
            <span style="color:#7a8099;font-size:11px;font-family:monospace;">${(e.received_at || '').slice(0,16)}</span>
          </div>
        </div>`;
    }).join('');

    document.getElementById('webhookEndpoint').textContent =
      window.location.origin + '/api/webhook/orderstatus';

  } catch(e) {
    console.error('Webhook stats error:', e);
  }
}
